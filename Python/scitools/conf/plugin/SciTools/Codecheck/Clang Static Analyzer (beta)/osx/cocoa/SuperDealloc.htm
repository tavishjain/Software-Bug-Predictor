1.1.8.20. osx.cocoa.SuperDealloc (ObjC)
Warn about improper use of '[super dealloc]' in Objective-C.

Examples: 

@interface SuperDeallocThenReleaseIvarClass : NSObject {
  NSObject *_ivar;
}
@end

@implementation SuperDeallocThenReleaseIvarClass
- (void)dealloc {
  [super dealloc];
  [_ivar release]; // warn
}
@end
